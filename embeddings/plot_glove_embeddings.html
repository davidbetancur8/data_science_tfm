def convert_text(t):
    return f'<b style="font-size: 20px"> {t} </b>'
In [8]:
set_names = set(names.flatten())
dict_names = {}
for name in set_names:
    r = randint(0, 255)
    g = randint(0, 255)
    b = randint(0, 255)
    dict_names[name] = f"rgb({r}, {g}, {b})"

colores = []
textos = []
cont = 0
names_f = names.flatten()
for tok in names_f:
    colores.append(dict_names[tok]) # colores
    if cont == 0:
        textos.append(f"{convert_text(names_f[cont])} {names_f[cont+1]}")
    elif cont == len(names_f)-1:
        textos.append(f"{names_f[cont-1]} {convert_text(names_f[cont])}")
    else:
        textos.append(f"{names_f[cont-1]} {convert_text(names_f[cont])} {names_f[cont+1]}")
    cont += 1
