def add_text_start(names_f, cont):
    texto = ""
    for i in range(6):
        if i == 0:
            texto = texto + (f" {convert_text(names_f[cont+i])}")
        else:
            texto = texto + (f" {names_f[cont+i]}")
    return texto

def add_text_end(names_f, cont):
    texto = ""
    for i in range(-5,1):
        if i == 0:
            texto = texto + (f" {convert_text(names_f[cont+i])}")
        else:
            texto = texto + (f" {names_f[cont+i]}")
    return texto

def add_text_inside(names_f, cont):
    texto = ""
    for i in range(-5,6):
        if i == 0:
            texto = texto + (f" {convert_text(names_f[cont+i])}")
        else:
            texto = texto + (f" {names_f[cont+i]}")
    return texto
